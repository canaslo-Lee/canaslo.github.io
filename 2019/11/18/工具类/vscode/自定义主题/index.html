自定义vscode主题

打开配置粘贴以下代码

// 自定义页面主题dark+ (default dark)
    "workbench.colorCustomizations": {
        "tab.activeBackground": "#118511",
        "activityBar.background": "#1b9b36a8",
        "sideBar.background": "#14412f94",
    },
    "editor.tokenColorCustomizations": {
        "tab.activeBackground": "#4fc26c",
        "activityBar.background": "#020000",
        "sideBar.background": "#000000",
        "[Default Dark+]": {
            //注释颜色
            "comments": "#48f8cf8c",
            //函数颜色
            "functions": "#e62df7",
            //关键字颜色
            "keywords": "#a31cfd",
            //数值颜色
            "numbers": "#ee6969",
            //字符串颜色
            "strings": "#8c7eda",
            //类型颜色
            "types": "#4400ff",
            //这个忘了 但在文件内有提示
            "variables": "#00ff2a",
            "textMateRules": [
                {
                    "scope": "keyword.operator",
                    "settings": {
                        "foreground": "#eeff00"
                    }
                }
            ]
        }
    }
